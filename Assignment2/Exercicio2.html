b- Em qual regime essa rede se encontra, subcritical, critical, supercritical ou conected? Justifique.

Para determinar em qual regime a rede se encontra (subcrtico, crtico, supercrtico ou conectado), podemos utilizar a relao do grau mdio com as fases de uma rede.

K < 1 : Subcritical
K == 1 : Critical
K > 1 : Supercritical
K > ln N : Connected

elif K == 1:
    condition = "Critical"
elif K > 1:
    condition = "Supercritical"
if K > np.log(N):
    condition = "Connected"

import numpy as np
import matplotlib.pyplot as plt

G_challenge = nx.read_gml("GraphMissingEdges.gml")
degree_sequence_challenge = [d for n, d in G_challenge.degree()]
degree_counts_challenge = np.bincount(degree_sequence_challenge)
degree_distribution_challenge = degree_counts_challenge / degree_counts_challenge.sum()

# Plotar a distribuio de graus da rede fornecida

plt.plot(range(len(degree_distribution_challenge)), degree_distribution_challenge)
plt.title("Distribuio de Graus da GraphMissingEdges")
plt.xlabel("Grau")
plt.ylabel("Probabilidade")
plt.grid(True)
plt.show()

from math import factorial
# Carregar a rede do desafio
G_challenge = nx.read_gml("GraphMissingEdges.gml")
N = len(G_challenge.nodes())
avg_degree = np.mean([d for n, d in G_challenge.degree()])

# Coeficiente de clusterizao
avg_cluster_coef = nx.average_clustering(G_challenge)
expected_cluster_coef = avg_degree / N
print("Coeficiente de clusterizao na rede do desafio:", avg_cluster_coef)
print("Coeficiente de clusterizao esperado para uma rede aleatria:", expected_cluster_coef)

# Distribuio de graus
degree_sequence_challenge = [d for n, d in G_challenge.degree()]
degree_counts_challenge = np.bincount(degree_sequence_challenge)
degree_distribution_challenge = degree_counts_challenge / degree_counts_challenge.sum()

k_values = np.arange(len(degree_distribution_challenge))
k_values_factorials = [factorial(value) for value in k_values]
expected_degree_distribution = np.exp(-avg_degree) * np.power(avg_degree, k_values) / (k_values_factorials)
expected_degree_distribution /= expected_degree_distribution.sum()

# Plotar a distribuio de graus da rede do desafio e a distribuio esperada para uma rede aleatria
plt.figure(figsize=(10, 6))
plt.plot(k_values, degree_distribution_challenge, marker='o', linestyle='-', color='b', label='Rede do Desafio')
plt.plot(k_values, expected_degree_distribution, linestyle='--', color='r', label='Rede Aleatria Esperada')
plt.title("Distribuio de Graus da Rede do Desafio e Rede Aleatria Esperada")

plt.ylabel("Probabilidade")
